-- Tipo de dato de las columnas de una tabla
DESCRIBE episodios;

-- Seleccionar todos los episodios donde la fecha de estreno sea mayor a 2010-01-01
SELECT
    *
FROM
    episodios
WHERE
    CAST(fecha_estreno AS DATE) > '2010-01-01';
