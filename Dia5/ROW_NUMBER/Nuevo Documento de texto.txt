-- top 3 de series mas recientes
WITH OrdenSeries AS (
SELECT
    titulo ,
    año_lanzamiento ,
    ROW_NUMBER() OVER (ORDER BY año_lanzamiento DESC) AS orden_lanzamiento
FROM series
)

SELECT * FROM OrdenSeries
WHERE orden_lanzamiento IN (1,2,3)

-- Consulta que seleccione todas las series, y utiliza
-- Row Number para asignar un numero secuencial basado en
-- su año de lanzamiento.
SELECT
    titulo ,
    año_lanzamiento ,
    ROW_NUMBER() over (ORDER BY año_lanzamiento DESC) AS clasificacion_global
FROM series
